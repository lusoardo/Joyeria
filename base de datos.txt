
create database Joyeria;
use Joyeria;
CREATE TABLE Cliente (
ID_Cliente int(6) PRIMARY KEY NOT NULL UNIQUE AUTO_INCREMENT,
Nombre varchar(15),
Apellido varchar(30),
Telefono int (6)
);
CREATE TABLE Pedido (
ID_Pedido int(6) PRIMARY KEY NOT NULL UNIQUE AUTO_INCREMENT,
Fecha date,
Precio int(6),
ID_Cliente int(6),
FOREIGN KEY(ID_Cliente) REFERENCES Cliente(ID_Cliente)
);
CREATE TABLE Producto (
ID_Producto int(6) PRIMARY KEY NOT NULL UNIQUE AUTO_INCREMENT,
Precio int(6),
Tamaño varchar (10),
 Oro varchar(10), 
 Plata varchar(10)
);
CREATE TABLE Tiene (
ID_Pedido int(6),
ID_Producto int(6),
FOREIGN KEY(ID_Pedido) REFERENCES Pedido(ID_Pedido),
FOREIGN KEY(ID_Producto) REFERENCES Producto(ID_Producto)

);
